 17 var Webview,
 18     utils = require("./utils"),
 19     webkitEvent = require('./webkitEvent'),
 20     qnx = utils.getQnxNamespace(),
 21     _self;

 23 /**
 24  * @class A javascript abstraction of the underlying webkit webview
 25  * @property {Number} id The id of the webview. READ ONLY
 26  * @property {Number} windowGroup The windowGroup of the webview. READ ONLY
 27  * @property {Boolean} visible Whether the webview is currently visible
 28  * @property {Boolean} active Whether the webview is currently active
 29  * @property {Number} zOrder The zOrder of the webview
 30  */
 31 function Webview() {
 32 
 33     var chromeId = 1,
 34         windowGroup = qnx.callExtensionMethod("webview.applicationWindowGroup", chromeId),
 35         id = parseInt(qnx.callExtensionMethod('webview.create', windowGroup, "InProcess"), 10),
 36         visible = false,
 37         active = false,
 38         zOrder = -99;
 39 
 40     this.__defineGetter__("id", function () {
 41         return id;
 42     });
 43 
 44     this.__defineGetter__("windowGroup", function () {
 45         return windowGroup;
 46     });
 47 
 48     this.__defineGetter__("visible", function () {
 49         return visible;
 50     });
 51 
 52     this.__defineSetter__("visible", function (newVisibility) {
 53         visible = !!newVisibility;
 54         qnx.callExtensionMethod("webview.setVisible", id, visible);
 55     });
 56 
 57     this.__defineGetter__("active", function () {
 58         return active;
 59     });
 60 
 61     this.__defineSetter__("active", function (newActivity) {
 62         active = !!newActivity;
 63         qnx.callExtensionMethod("webview.setActive", id, active);
 64     });
 65 
 66     this.__defineGetter__("zOrder", function () {
 67         return zOrder;
 68     });
 69 
 70     this.__defineSetter__("zOrder", function (newZOrder) {
 71         zOrder = parseInt(newZOrder, 10);
 72         qnx.callExtensionMethod("webview.setZOrder", id, zOrder);
 73     });
 74     
 75 }

 77 /**#@+
 78 * @param {String} eventType The native event to be listened to. 
 79 * Can be one of 'PropertyViewportEvent', 'QNXWebDestroyedEvent', 'Destroyed', 'Created', 'PropertyLoadProgressEvent', 'PropertyLocationEvent', 'PropertyTitleEvent', 'PropertyCanGoBackEvent', 'PropertyCanGoForwardEvent', 'PropertyFaviconEvent', 'PropertySecureTypeEvent', 'JavaScriptResult', 'ContentRendered', 'JavaScriptWindowObjectCleared', 'PropertyTooltipEvent', 'Created', 'DocumentLoadCommitted', 'DocumentLoaded', 'DocumentLoadFinished', 'LocationChange', 'LocationChanging', 'NetworkError', 'PropertyActiveEvent', 'PropertyBackgroundColorEvent', 'PropertyCertificateInfoEvent', 'PropertyContentRectangleEvent', 'PropertyEnableWebInspectorEvent', 'PropertyEncryptionInfoEvent', 'PropertyHistoryListEvent', 'PropertyHistoryPositionEvent', 'PropertyJavaScriptInterruptTimeoutEvent', 'PropertyOriginalLocationEvent', 'PropertyScaleEvent', 'PropertyScrollPositionEvent', 'PropertyStatusEvent', 'PropertyVisibleEvent', 'PropertyWebInspectorPortEvent', 'PropertyZOrderEvent'.
 80 **/

 82 /**
 83  * @description Adds a listener for a given event. Multiple listeners can be registered for the same event.
 84  * @param {callback} eventListener The function to be invoked when the event occurs
 85  */
 86 Webview.prototype.addEventListener = function (eventType, eventListener) {
 87     webkitEvent.on({id: this.id, eventType: eventType}, eventListener); //What to put for scope???
 88 };
 89 
 90 /**
 91  * @description Removes a specific registered listener for a given event
 92  * @param {Object} eventListener The function to be removed from the list of event listeners
 93  */
 94 Webview.prototype.removeEventListener = function (eventType, eventListener) {
 95     webkitEvent.removeEventListener({id: this.id, eventType: eventType}, eventListener);
 96 };
 97 
 98 /**
 99  * @description Removes all registered listeners for a given event
100  */
101 Webview.prototype.removeAllEventListeners = function (eventType) {
102     webkitEvent.clear({id: this.id, eventType: eventType});
103 };
104 
105 /**
106  * @description Dispatches an event of the provided type with the given arguments to all registered listeners.
107  * @param {Object[]} args The array of objects to be passed to the event
108  */
109 Webview.prototype.dispatchEvent = function (eventType, args) {
110     webkitEvent.emit({id: this.id, eventType: eventType}, args);
111 };

115 /**
116 * @description Sets the location of the webview on the screen
117 * @param {Number} x The x position of the webview
118 * @param {Number} y The y position of the webview
119 * @param {Number} width The width of the webview
120 * @param {Number} height The height of the webview
121 */
122 Webview.prototype.setGeometry = function (x, y, width, height) {
123     qnx.callExtensionMethod("webview.setGeometry", this.id, x, y, width, height);
124 };
125 
126 /**
127 * @description Destroys the underlying native webview
128 * @param {callback} [onComplete] A callback to be invoked when the destruction call is complete
129 */ 
130 Webview.prototype.destroy = function (onComplete) {
131     qnx.callExtensionMethod("webview.destroy", this.id);
132 
133     if (onComplete && typeof onComplete === 'function') {
134         onComplete();
135     }
136 };
137 
138 /**
139 * @description Sets the URL on the webview
140 * @param {String} url The fully qualified url to be loaded into the webview
141 */ 
142 Webview.prototype.setURL = function (url) {
143     qnx.callExtensionMethod("webview.loadURL", this.id, url);
144 };
145 
146 /**
147 * @description Executes javascript in the context of the webview
148 * @param {String} js The javascript expression to be executed
149      * @param {boolean} [inIsolatedWorld=false] Run javascript in isolated context
150      * @param {callback} [onComplete] A callback to be invoked when the destruction call is complete
151 */ 
152 Webview.prototype.executeJavaScript = function (js, inIsolatedWorld, onComplete) {
153     qnx.callExtensionMethod("webview.executeJavaScript", this.id, js, inIsolatedWorld || false);
154 
155     if (onComplete && typeof onComplete === 'function') {
156         onComplete();
157     }
158 };

160 /**
161      * @description Sets the background color of the webview
162      * @param {String} color The desired background color of the webview in hex
163      * @example webview.setBackgroundColor("0x00FFFFFF");
164 */
165 Webview.prototype.setBackgroundColor = function (color) {
166     qnx.callExtensionMethod("webview.setBackgroundColor", this.id, color);
167 };
