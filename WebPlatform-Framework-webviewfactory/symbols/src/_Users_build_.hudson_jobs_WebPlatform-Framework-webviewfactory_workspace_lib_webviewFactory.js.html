 17 var Webview,
 18     utils = require("./utils"),
 19     webkitEvent = require('./webkitEvent'),
 20     qnx = utils.getQnxNamespace(),
 21     _self;
 22 
 23 /**
 24  * @class A javascript abstraction of the underlying webkit webview
 25  * @property {Number} id The id of the webview. READ ONLY
 26  * @property {Number} windowGroup The windowGroup of the webview. READ ONLY
 27  * @property {Boolean} visible Whether the webview is currently visible
 28  * @property {Boolean} active Whether the webview is currently active
 29  * @property {Number} zOrder The zOrder of the webview
 30  * @property {String} url The fully qualified url currently loaded into the webview
 31  */
 32 function Webview() {
 33 
 34     var chromeId = 1,
 35         windowGroup = qnx.callExtensionMethod("webview.applicationWindowGroup", chromeId),
 36         id = parseInt(qnx.callExtensionMethod('webview.create', windowGroup, "InProcess"), 10),
 37         visible = false,
 38         active = false,
 39         zOrder = -99,
 40         url = "";
 41 
 42     this.__defineGetter__("id", function () {
 43         return id;
 44     });
 45 
 46     this.__defineGetter__("windowGroup", function () {
 47         return windowGroup;
 48     });
 49 
 50     this.__defineGetter__("visible", function () {
 51         return visible;
 52     });
 53 
 54     this.__defineSetter__("visible", function (newVisibility) {
 55         visible = !!newVisibility;
 56         qnx.callExtensionMethod("webview.setVisible", id, visible);
 57     });
 58 
 59     this.__defineGetter__("active", function () {
 60         return active;
 61     });
 62 
 63     this.__defineSetter__("active", function (newActivity) {
 64         active = !!newActivity;
 65         qnx.callExtensionMethod("webview.setActive", id, active);
 66     });
 67 
 68     this.__defineGetter__("zOrder", function () {
 69         return zOrder;
 70     });
 71 
 72     this.__defineSetter__("zOrder", function (newZOrder) {
 73         zOrder = parseInt(newZOrder, 10);
 74         qnx.callExtensionMethod("webview.setZOrder", id, zOrder);
 75     });
 76     
 77     this.__defineGetter__("url", function () {
 78         return url;
 79     });
 80 
 81     this.__defineSetter__("url", function (newUrl) {
 82         url = newUrl;
 83         qnx.callExtensionMethod("webview.loadURL", this.id, url);
 84     });
 85 
 86     webkitEvent.on(this.id, "LocationChange", function (newUrl) {
 87         this.url = newUrl;
 88     }, this); //send this for scope so it can update the url property
 89 }
 90 
 91 /**#@+
 92 * @param {String} eventType The native event to be listened to. 
 93 * Can be one of 'PropertyViewportEvent', 'QNXWebDestroyedEvent', 'Destroyed', 'Created', 'PropertyLoadProgressEvent', 'PropertyLocationEvent', 'PropertyTitleEvent', 'PropertyCanGoBackEvent', 'PropertyCanGoForwardEvent', 'PropertyFaviconEvent', 'PropertySecureTypeEvent', 'JavaScriptResult', 'ContentRendered', 'JavaScriptWindowObjectCleared', 'PropertyTooltipEvent', 'Created', 'DocumentLoadCommitted', 'DocumentLoaded', 'DocumentLoadFinished', 'LocationChange', 'LocationChanging', 'NetworkError', 'PropertyActiveEvent', 'PropertyBackgroundColorEvent', 'PropertyCertificateInfoEvent', 'PropertyContentRectangleEvent', 'PropertyEnableWebInspectorEvent', 'PropertyEncryptionInfoEvent', 'PropertyHistoryListEvent', 'PropertyHistoryPositionEvent', 'PropertyJavaScriptInterruptTimeoutEvent', 'PropertyOriginalLocationEvent', 'PropertyScaleEvent', 'PropertyScrollPositionEvent', 'PropertyStatusEvent', 'PropertyVisibleEvent', 'PropertyWebInspectorPortEvent', 'PropertyZOrderEvent'.
 94 **/
 95 
 96 /**
 97  * @description Adds a listener for a given event. Multiple listeners can be registered for the same event.
 98  * @param {callback} eventListener The function to be invoked when the event occurs
 99  */
100 Webview.prototype.addEventListener = function (eventType, eventListener) {
101     webkitEvent.on(this.id, eventType, eventListener); //What to put for scope???
102 };
103 
104 /**
105  * @description Removes a specific registered listener for a given event
106  * @param {Object} eventListener The function to be removed from the list of event listeners
107  */
108 Webview.prototype.removeEventListener = function (eventType, eventListener) {
109     webkitEvent.removeEventListener(this.id, eventType, eventListener);
110 };
111 
112 /**
113  * @description Dispatches an event of the provided type with the given arguments to all registered listeners.
114  * @param {Object[]} args The array of objects to be passed to the event
115  */
116 Webview.prototype.dispatchEvent = function (eventType, args) {
117     webkitEvent.emit(this.id, eventType, args);
118 };

122 /**
123 * @description Sets the location of the webview on the screen
124 * @param {Number} x The x position of the webview
125 * @param {Number} y The y position of the webview
126 * @param {Number} width The width of the webview
127 * @param {Number} height The height of the webview
128 */
129 Webview.prototype.setGeometry = function (x, y, width, height) {
130     qnx.callExtensionMethod("webview.setGeometry", this.id, x, y, width, height);
131 };
132 
133 /**
134 * @description Destroys the underlying native webview
135 * @param {callback} [onComplete] A callback to be invoked when the destruction call is complete
136 */ 
137 Webview.prototype.destroy = function (onComplete) {
138     qnx.callExtensionMethod("webview.destroy", this.id);
139     webkitEvent.clear(this.id);
140     if (onComplete && typeof onComplete === 'function') {
141         onComplete();
142     }
143 };
144 
145 /**
146 * @description Executes javascript in the context of the webview
147 * @param {String} js The javascript expression to be executed
148      * @param {boolean} [inIsolatedWorld=false] Run javascript in isolated context
149      * @param {callback} [onComplete] A callback to be invoked when the destruction call is complete
150 */ 
151 Webview.prototype.executeJavaScript = function (js, inIsolatedWorld, onComplete) {
152     qnx.callExtensionMethod("webview.executeJavaScript", this.id, js, inIsolatedWorld || false);
153 
154     if (onComplete && typeof onComplete === 'function') {
155         onComplete();
156     }
157 };
158 
159 /**
160      * @description Sets the background color of the webview
161      * @param {String} color The desired background color of the webview in hex
162      * @example webview.setBackgroundColor("0x00FFFFFF");
163 */
164 Webview.prototype.setBackgroundColor = function (color) {
165     qnx.callExtensionMethod("webview.setBackgroundColor", this.id, color);
166 };
