 17 var evt = require('./webkitEvent'),
 18     message = require('./message'),
 19     request = require('./request'),
 20     onRequest,
 21     chromeId = 1,
 22     chromeHeight = 0,
 23     windowGroup = -1,
 24     id = -1,
 25     _self;
 26 
 27 evt.on("NetworkResourceRequested", function (args) {
 28     var req = request.init(args);
 29     if (onRequest) {
 30         onRequest(req);
 31     }
 32     else {
 33         req.allow();
 34     }
 35 });
 36 
 37 /**
 38  * @namespace webview
 39  * @exports _self as webview
 40  */
 41 _self = 
 42     {
 43 
 44     /**
 45      * @param {callback} ready A callback invoked when the webview is created
 46      */ 
 47     create: function (ready, configSettings) {
 48         message.init();
 49         qnx.callExtensionMethod("webview.setEnableWebInspector", chromeId, (configSettings ? (!!configSettings.debugEnabled) : false));
 50         qnx.callExtensionMethod("webview.setEnableCrossSiteXHR", chromeId, "true");
 51         qnx.callExtensionMethod("webview.setVisible", chromeId, "true");
 52         qnx.callExtensionMethod("webview.setActive", chromeId, "false");
 53         qnx.callExtensionMethod("webview.setGeometry", chromeId, 0, 0, screen.width, chromeHeight);
 54 
 55         windowGroup = qnx.callExtensionMethod("webview.applicationWindowGroup", chromeId);
 56 
 57         id = qnx.callExtensionMethod("webview.create", windowGroup);
 58 
 59         qnx.callExtensionMethod("webview.setVisible", id, true);
 60         qnx.callExtensionMethod("webview.setActive", id, true);
 61         qnx.callExtensionMethod("webview.setZOrder", id, 0);
 62         qnx.callExtensionMethod("webview.setEnableNetworkResourceRequestedEvents", id, true);
 63         qnx.callExtensionMethod("webview.setGeometry", id, 0, chromeHeight, screen.width, screen.height - chromeHeight);
 64         qnx.callExtensionMethod("applicationWindow.setVisible", "true");
 65 
 66         if (ready && typeof ready === 'function') {
 67             ready();
 68         }
 69     },
 70 
 71     destroy: function () {
 72         qnx.callExtensionMethod("webview.destroy", id);
 73     },
 74 
 75     /**
 76      * @param {String} url The fully qualified url to be loaded into the webview
 77      */ 
 78     setURL: function (url) {
 79         qnx.callExtensionMethod("webview.loadURL", id, url);
 80     },
 81 
 82     /**
 83      * @event
 84      * @param {callback} callback The method to be invoked when a resource is requested
 85      */ 
 86     onRequest: function (callback) {
 87         if (typeof callback === "function" || callback === null) {
 88             onRequest = callback;
 89         }
 90     },
 91 
 92     /**
 93      * @param {String} js The javascript expression to be executed
 94      */ 
 95     executeJavascript: function (js) {
 96         qnx.callExtensionMethod("webview.executeJavaScript", id, js, false);
 97     }
 98 };
 99 
100 module.exports = _self;
