 18 /**
 19  * @exports _self as qnx.webplatform
 20  * @namespace qnx.webplatform
 21 */
 22 qnx.webplatform = (function () {
 23     var _r = window.require,
 24         WebView,
 25         ApplicationWindow,
 26         chrome,
 27         _controller,
 28         _applicationWindow,
 29         internal;
 30         
 31     window.require = function (id) {
 32         id = id.replace(/^\.\.\/\.\.\//, "");
 33         return _r(id);
 34     };
 35         
 36     WebView = require("WebView");
 37     chrome = require("chrome");
 38     internal = require("chrome/internal");
 39     ApplicationWindow = require("ApplicationWindow");
 40 
 41     _controller = new WebView({WebViewId : chrome.id});
 42     _applicationWindow = new ApplicationWindow();
 43     
 44     window.chrome.internal.webEvent = internal.webEvent;
 45 
 46     return {
 47         /**
 48          * @description This method creates a new webview object 
 49          * @param {Function} onCreate A callback that will be fired when the native side creates the webview. Clients are expected to implement this as the native side will throw errors if the user attempts to access the webview before the event occures.
 50          * @returns {WebView} A webview object
 51          * @example
 52          * qnx.webplatform.getController().enableWebInspector(true);
 53          *
 54          * var webview = qnx.webplatform.createWebView(function (value, eventId) {
 55          *     webview.setGeometry(0, 0, screen.width, screen.height);
 56          *     webview.visible = true;
 57          *     webview.active = true;
 58          *     console.log("woot");
 59          * });
 60          */ 
 61         createWebView : function (onCreate) {
 62             return new WebView({onCreate : onCreate});
 63         },
 64         
 65         /**
 66          * @description This method returns the Webview object for the controller
 67          * @returns {WebView} The controller webview
 68          */ 
 69         getController : function () {
 70             return _controller;
 71         },
 72 
 73         /**
 74          * @description This method returns the ApplicationWindow object
 75          * @returns {ApplicationWindow} The Application Window
 76          */ 
 77         getApplicationWindow : function () {
 78             return _applicationWindow;
 79         }
 80     };
 81 }());
