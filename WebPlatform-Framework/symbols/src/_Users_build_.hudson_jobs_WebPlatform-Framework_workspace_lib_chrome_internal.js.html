  4 var events = require('../events'),
  5     chrome = require("../chrome");
  6 
  7 module.exports = {
  8     webEvent: function (id, eventId, value) {
  9         switch (eventId) {
 10         case 'PropertyViewportEvent':
 11         case 'QNXWebDestroyedEvent':
 12         case 'Destroyed':
 13         case 'Created':
 14         case 'PropertyLoadProgressEvent':
 15         case 'PropertyLocationEvent':
 16         case 'PropertyTitleEvent':
 17         case 'PropertyCanGoBackEvent':
 18         case 'PropertyCanGoForwardEvent':
 19         case 'PropertyFaviconEvent':
 20         case 'PropertySecureTypeEvent':
 21         case 'JavaScriptResult':
 22         case 'ContentRendered':
 23         case 'JavaScriptWindowObjectCleared':
 24         case 'PropertyTooltipEvent':
 25         case 'Created':
 26         case 'DocumentLoadCommitted':
 27         case 'DocumentLoaded':
 28         case 'DocumentLoadFinished':
 29         case 'LocationChange':
 30         case 'LocationChanging':
 31         case 'NetworkError':
 32         case 'NetworkResourceRequested':
 33         case 'PropertyActiveEvent':
 34         case 'PropertyBackgroundColorEvent':
 35         case 'PropertyCertificateInfoEvent':
 36         case 'PropertyContentRectangleEvent':
 37         case 'PropertyEnableWebInspectorEvent':
 38         case 'PropertyEncryptionInfoEvent':
 39         case 'PropertyHistoryListEvent':
 40         case 'PropertyHistoryPositionEvent':
 41         case 'PropertyJavaScriptInterruptTimeoutEvent':
 42         case 'PropertyOriginalLocationEvent':
 43         case 'PropertyScaleEvent':
 44         case 'PropertyScrollPositionEvent':
 45         case 'PropertyStatusEvent':
 46         case 'PropertyVisibleEvent':
 47         case 'PropertyWebInspectorPortEvent':
 48         case 'PropertyZOrderEvent':
 49             events.emit(id, eventId, [value]);
 50             break;
 51         default:
 52             qnx.callExtensionMethod("webview.printToStderr", chrome.id, "Unknown Event: " + eventId + ":" + value + "\n");
 53             break;
 54         }
 55     }
 56 };
