 17 var ApplicationWindow;
 18 
 19 /**
 20  * @class A javascript abstraction of the application window
 21  * @property {Boolean} visible Whether the application window is visible or not
 22  */
 23 ApplicationWindow = function () {
 24 
 25     this.__defineGetter__("visible", function () {
 26         return !!qnx.callExtensionMethod("applicationWindow.isVisible");
 27     });
 28 
 29     this.__defineSetter__("visible", function (newVisibility) {
 30         qnx.callExtensionMethod("applicationWindow.setVisible", !!newVisibility);
 31     });
 32 };
 33 
 34 module.exports = ApplicationWindow;
