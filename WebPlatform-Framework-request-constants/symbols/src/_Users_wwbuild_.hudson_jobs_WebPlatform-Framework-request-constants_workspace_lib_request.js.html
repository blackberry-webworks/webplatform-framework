 16 var evt = require('./webkitEvent'),
 17     message = require('./message'),
 18     exception = require('./exception'),
 19     _self;
 20 
 21 evt.on("NetworkResourceRequestedResponse", function (args) {
 22     qnx.callExtensionMethod("webview.notifyOpen", args.id, args.sid, args.response.code, "OK");
 23     qnx.callExtensionMethod("webview.notifyHeaderReceived", args.id, args.sid, "Access-Control-Allow-Origin", "*");
 24     qnx.callExtensionMethod("webview.notifyDataReceived", args.id, args.sid, args.response.responseText, args.response.responseText.length);
 25     qnx.callExtensionMethod("webview.notifyDone", args.id, args.sid);
 26 });
 27 
 28 _self = {
 29     init: function (args) {
 30         var takeover = false;
 31         return {
 32             url: args.data.url,
 33             //TODO: this origin won't work for iframes
 34             origin: qnx.callExtensionMethod("webview.originalLocation", args.id),
 35             //TODO: body
 36             //body: args.body,
 37             allow: function () {
 38                 message.send("NetworkResourceRequestedResponse", {
 39                     id: args.id,
 40                     url: args.data.url,
 41                     response: {
 42                         setAction: "ACCEPT"
 43                     }
 44                 });
 45             },
 46             deny: function () {
 47                 message.send("NetworkResourceRequestedResponse", {
 48                     id: args.id,
 49                     url: args.data.url,
 50                     response: {
 51                         setAction: "DENY"
 52                     }
 53                 });
 54             },
 55             substitute: function () {
 56                 takeover = true;
 57                 message.send("NetworkResourceRequestedResponse", {
 58                     id: args.id,
 59                     url: args.data.url,
 60                     response: {
 61                         setAction: "SUBSTITUTE"
 62                     }
 63                 });
 64             },
 65             respond: function (code, responseText) {
 66                 if (takeover) {
 67                     message.send("NetworkResourceRequestedResponse", {
 68                         id: args.id,
 69                         sid: args.data.streamId,
 70                         url: args.data.url,
 71                         response: {
 72                             code: code,
 73                             responseText: responseText
 74                         }
 75                     }, true);
 76                 }
 77                 else {
 78                     exception.raise(exception.types.InvalidState, "Cannot respond until after substitute has been requested");
 79                 }
 80             }
 81         };
 82     }
 83 };
 84 
 85 module.exports = _self;
